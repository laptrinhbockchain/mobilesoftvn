package com.mobilesoftvn.testingassistant.app;

import java.util.ArrayList;

public class DataUtils {
	public static boolean isSame(ArrayList<Integer> obj1, ArrayList<Integer> obj2) {
		if (obj1==null || obj2==null) return false;
		if (obj1.size()==0 || obj2.size()==0) return false;
		
		boolean ret1 = true;
		int num1 = obj1.size();
		for (Integer i=0; i<num1; i++) {
			if (!obj2.contains(obj1.get(i))) {
				ret1 = false;
				break;
			}
		}
		
		boolean ret2 = true;
		int num2 = obj2.size();
		for (Integer i=0; i<num2; i++) {
			if (!obj1.contains(obj2.get(i))) {
				ret2 = false;
				break;
			}
		}

		return (ret1 && ret2);
	}
}
